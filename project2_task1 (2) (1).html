#defining nvg_hvg function
def NVG_HVG(method):
    
    #Horizontal visibility graphs (HVGs) are graphs generated from time series data    
    if method == 'HVG':
        for i in axes:
            a = HorizontalVG()
            a.build(data_hc[i])
            ig_g = a.as_igraph()
            
            #calculations
            avg_degree = np.mean(ig_g.degree())
            network_dia = ig_g.diameter()
            path_length = ig_g.average_path_length()

            degree.append(avg_degree)
            diameter.append(network_dia)
            length.append(path_length)
            axis.append(i)
            methods.append(method)
    #natural visibility graph 
    elif method== 'NVG':
        for i in axes:
            a = NaturalVG()
            a.build(data_hc[i])
            ig_g = a.as_igraph()
            
            #calculations
            avg_degree = np.mean(ig_g.degree())
            network_dia = ig_g.diameter()
            path_length = ig_g.average_path_length()

            degree.append(avg_degree)
            diameter.append(network_dia)
            length.append(path_length)
            axis.append(i)
            methods.append(method)
